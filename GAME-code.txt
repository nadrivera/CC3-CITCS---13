    public void animate() {
        y = (y + 1) % 2;

        switch (direction) {

            case KeyEvent.VK_DOWN:
                //bed
                if (xp <= 40 && yp <= 35) {
                    health = 20;
                    statusHealth = "HP: " + health;
                }
                x = 1;
                //mother's walls
                if ((yp >= 170 && yp <= 360) && xp <= 140) {
